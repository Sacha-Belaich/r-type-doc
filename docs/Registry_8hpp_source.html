24 template <class Component>
25 SparseArray<Component> &register_component() {
26 const std::type_index idx(typeid(Component));
27 if (!_components_arrays.contains(idx)) {
28 _components_arrays.emplace(idx, SparseArray<Component>());
29 }
30 if (!_remove_functions.contains(idx)) {
31 _remove_functions.emplace(idx, [](Registry &reg, Entity const &ent) {
32 auto &array = reg.get_components<Component>();
33 array.erase(ent);
34 });
35 }
36 if (!_loggers.contains(idx)) {
37 _loggers.emplace(idx, [](Registry const &reg, Entity const &ent) {
38 const auto &array = reg.get_components<Component>();
39 const std::optional<Component> &comp = array[ent];
40 if (comp.has_value()) comp.value().log();
41 return comp.has_value();
42 });
43 }
44 return std::any_cast<SparseArray<Component> &>(_components_arrays[idx]);
45 }
